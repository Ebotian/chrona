#!/usr/bin/env bash

pidwait() {
    n=0
    echo $pid
    while true; do
        if [ $(ps mux | grep "$pid" | wc -l) -eq 3 ]; then
            echo "Process $pid done"
            exit 0
        fi
        echo 'pid-----'
        echo $(ps mux | grep "$pid" | wc -l)
        echo '2-----'
        echo $pid
        echo '3-----'
        sleep 1
        echo "do $n seconds"
        echo '4-----'
        let n++
    done
}
